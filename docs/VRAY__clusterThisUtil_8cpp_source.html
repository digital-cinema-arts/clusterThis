107 void VRAY_clusterThis::convert(
108  openvdb::ScalarGrid::Ptr outputGrid,
109  ParticleList & paList,
110  const Settings & settings,
111  hvdb::Interrupter & boss)
112 {
113 
114  openvdb::tools::ParticlesToLevelSet<openvdb::ScalarGrid, ParticleList, hvdb::Interrupter> raster(*outputGrid, boss);
115 
116  if(myVerbose == CLUSTER_MSG_DEBUG)
117  std::cout << "VRAY_clusterThis::convert() " << std::endl;
118 
119  raster.setRmin(settings.mRadiusMin);
120 
121  if(myVerbose == CLUSTER_MSG_DEBUG) {
122  std::cout << "VRAY_clusterThis::convert(): raster.getVoxelSize(): " << raster.getVoxelSize() << std::endl;
123  std::cout << "VRAY_clusterThis::convert(): raster.getRmin(): " << raster.getRmin() << std::endl;
124  std::cout << "VRAY_clusterThis::convert(): raster.getHalfWidth(): " << raster.getHalfWidth() << std::endl;
125  }
126 
127  if(raster.getHalfWidth() < openvdb::Real(2)) {
128  std::cout << "VRAY_clusterThis::convert(): Half width of narrow-band < 2 voxels which creates holes when meshed!" << std::endl;
129  }
130  else
131  if(raster.getHalfWidth() > openvdb::Real(1000)) {
132  throw std::runtime_error(
133  "VRAY_clusterThis::convert(): Half width of narrow-band > 1000 voxels which exceeds memory limitations!");
134  }
135 
136  if(settings.mRasterizeTrails && paList.hasVelocity()) {
137  if(myVerbose == CLUSTER_MSG_DEBUG)
138  std::cout << "VRAY_clusterThis::convert(): rasterizing trails" << std::endl;
139  raster.rasterizeTrails(paList, settings.mDx);
140  }
141  else {
142  if(myVerbose == CLUSTER_MSG_DEBUG)
143  std::cout << "VRAY_clusterThis::convert(): rasterizing spheres" << std::endl;
144  raster.rasterizeSpheres(paList);
145  }
146 
147  if(boss.wasInterrupted()) {
148  if(myVerbose == CLUSTER_MSG_DEBUG)
149  std::cout << "VRAY_clusterThis::convert(): Process was interrupted" << std::endl;
150  return;
151  }
152 
153  // Convert the level-set into a fog volume.
154  if(settings.mFogVolume) {
155  if(myVerbose == CLUSTER_MSG_DEBUG)
156  std::cout << "VRAY_clusterThis::convert(): converting to fog volume" << std::endl;
157  float cutOffDist = std::numeric_limits<float>::max();
158  if(settings.mGradientWidth > 1e-6)
159  cutOffDist = settings.mGradientWidth;
160  openvdb::tools::levelSetToFogVolume(*outputGrid, cutOffDist, false);
161  }
162 
163 // print stats of the vdb grid
164  if(myVerbose == CLUSTER_MSG_DEBUG)
165  outputGrid->print();
166 
167 }
